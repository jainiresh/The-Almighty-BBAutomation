<script type="text/javascript">
    var tabs = document.getElementsByClassName('hiddenTab');
    var curTab = 0;
    
    var pauseTabs = false;
    
    function changeTabStatus(tabID) {
        
        myTab = document.getElementById(tabID);
        
        node = myTab.parentNode.firstElementChild;
        
        while( node && node.nodeType === 1 ) {
            node.className = 'inactive';
            node = node.nextElementSibling;
        }
        
        myTab.className = 'active';
        
        document.getElementById('tabsContent').innerHTML = document.getElementById(tabID + 'Content').innerHTML;
    }
    
    function rotateTabs() {
        if (!pauseTabs) {
            curTab++;
        
            if (curTab >= tabs.length) {
                curTab = 0;
            }
        
            changeTabStatus(tabs[curTab].id.substring(0, tabs[curTab].id.length - 7));
        }
    }
    
    function stopTabs() {
        pauseTabs = true;
    }
    
    function startTabs() {
        pauseTabs = false;
    }
</script>
