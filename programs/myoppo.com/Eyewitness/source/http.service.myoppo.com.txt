        // 封装 CryptJS的AES加密算法
        function encryptAES(data, keyString) {
            var key = CryptoJS.enc.Utf8.parse(keyString);
            var iv = CryptoJS.enc.Utf8.parse(keyString);
            var encrypted = CryptoJS.AES.encrypt(CryptoJS.enc.Utf8.parse(data), key,
                {
                    keySize: 128 / 8,
                    iv: iv,
                    mode: CryptoJS.mode.CBC,
                    padding: CryptoJS.pad.Pkcs7
                });

            return encrypted;
        }

        function editWindow(url, param, width, height) {
            ///	<summary>通过window.open的方式打开窗口,窗口关闭时，可以自动刷新父窗口 joesong 20121123</summary>
            ///	<param name="param" type="String">控制窗口的特性，默认保持为空即可</param>
            ///	<param name="param" type="Integer">窗口的宽度，如果不传值，则默认为 1000</param>
            ///	<param name="param" type="height">窗口的高度，如果不传值，则默认为 600</param>
            if (width == undefined)
                width = 1000;
            if (height == undefined)
                height = 600;

            if (param == undefined || param == "")
                param = ',toolbar=no,menubar=no,scrollbars=yes, resizable=yes,location=no,status=no,width=' + width + ',height=' + height;

            var win = window.open(url, '', param);

            var s_height = window.screen.availHeight;
            var s_width = window.screen.availWidth;
            var w_height = height;
            var w_width = width;
            if (w_height > (s_height - 30))
                w_height = s_height - 30;

            if (w_width > (s_width - 10))
                w_width = s_width - 10;

            var itop = (s_height - 30 - w_height) / 2; //获得窗口的垂直位置;
            var ileft = (s_width - 10 - w_width) / 2; //获得窗口的水平位置;
            win.moveTo(ileft, itop);
        }

                }
                $("#reload_verifycode").hide();
            });
        }

        // 发送短信验证码 开始倒计时
        var countdown;
        function settime(obj) {
            countdown = getCookieValue("secondsremained");
            if (countdown == 0) {
                obj.removeAttr("disabled");
                obj.val("获取验证码");
                return;
            } else {
                obj.attr("disabled", true);
                obj.val("重新获取(" + countdown + ")");
                countdown--;
                editCookie("secondsremained", countdown, countdown + 1);
            }
            setTimeout(function () { settime(obj) }, 1000) // 每1000毫秒执行一次
        }

        // 发送短信验证码 添加cookie
        function addCookie(name, value, expiresHours) {
            var cookieString = name + "=" + escape(value);
            // 判断是否设置过期时间,0代表关闭浏览器时失效
            if (expiresHours > 0) {
                var date = new Date();
                date.setTime(date.getTime() + expiresHours * 1000);
                cookieString = cookieString + ";expires=" + date.toUTCString();
            }
            document.cookie = cookieString;
        }

        // 发送短信验证码 修改cookie的值
        function editCookie(name, value, expiresHours) {
            var cookieString = name + "=" + escape(value);
            if (expiresHours > 0) {
                var date = new Date();
                date.setTime(date.getTime() + expiresHours * 1000); // 单位是毫秒
                cookieString = cookieString + ";expires=" + date.toGMTString();
            }
            document.cookie = cookieString;
        }

        // 发送短信验证码 根据名字获取cookie的值
        function getCookieValue(name) {
            var strCookie = document.cookie;
            var arrCookie = strCookie.split("; ");
            for (var i = 0; i < arrCookie.length; i++) {
                var arr = arrCookie[i].split("=");
                if (arr[0] == name) {
                    return unescape(arr[1]);
                    break;
                } else {
                    return "";
                    break;
                }
            }
        }
