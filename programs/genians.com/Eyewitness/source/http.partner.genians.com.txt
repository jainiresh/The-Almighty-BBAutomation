<script>
    var locationUrl = "/";
    var popup = null;

    var userLogin = {
        window: null,
        loginType: null,
        jira: function(){
            this.loginType = 'jira';
            this.fetchSsoLogin( res => {
                if(res.code === 0){
                    return location.replace(locationUrl);
                }

                if(res.code === 401){
                    Common.toast(res.description);
                    return;
                }

                if(res.code > 0){
                    this.window = window.open('https://sso.genians.com/login?redirectTo=https%3A%2F%2Fims.genians.com%2Fjira&external_login=PARTNER_PORTAL','JIRA_LOGIN','scrollbars=no,resizable=no,status=no,location=no,toolbar=no,menubar=no,width=600,height=500,left=-1000,top=-1000');
                }
            });
        },
        bizportal: function(){
            this.loginType = 'bizportal';

            this.fetchSsoLogin( res => {
                if(res.code === 0){
                    return location.replace(locationUrl);
                }

                if(res.code === 401){
                    Common.toast(res.description);
                    return;
                }

                if(res.code > 0){
                    const url = "https://bp.genians.com/bp/?mode=login&external_login=PARTNER_PORTAL";
                    this.window = window.open(url, 'BIZPORTAL_LOGIN', 'scrollbars=no,resizable=no,status=no,location=no,toolbar=no,menubar=no,width=600,height=500,left=-1000,top=-1000');
                }
            });
        },

        initWindowEvent: function(){
            window.addEventListener("message", function(event) {

                if (event.origin == window.location.origin) {
                    // something from an unknown domain, let's ignore it
                    return;
                }

                if (event.data == 'refresh') {
                    userLogin.fetchSsoLogin();
                }
            });
        },
        fetchSsoLogin: function(resultCallback){

            const ssoLoginUrl = '/public/user/api/ssoLogin/' + this.loginType;
            const fetcher = new Fetcher({
                url: ssoLoginUrl,
                data: {},
                success: res => {
                    if(resultCallback){
                        return resultCallback(res);
                    }

                    if(res.code === 0){
                        location.replace(locationUrl);
                    }else{
                        Common.toast(res.description);
                    }
                },
                error: err => {
                    if(!err.responseJSON){
                        return Common.toast('로그인 API 형식 오류');
                    }

                    if(resultCallback){
                        return resultCallback(err.responseJSON);
                    }

                    Common.toast(err.responseJSON.description);
                }
            });

            fetcher.post();
        },
    };

    $(document).ready( function(){
        userLogin.initWindowEvent();
    })


</script>
